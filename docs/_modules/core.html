[docs]def runV2GSim(vehicleList, nbIteration=1, option=None, itineraryNb=0):
    """ Launch driving functions and charging functions for each vehicle passed,
    results from function is then appended to the vehicle state.

    Args:
        vehicleList (list of vehicles): vehicles to simulate
        nbIteration (int): number of iterations
        option (anything): optional parameter for charging functions
        itineraryNb (int): optional itinerary number (default: 0)
    """
    print(colored.green('core.runV2GSim(vehicleList, nbIteration=' + str(nbIteration) + ')'))

    # Create the progress bar
    progress = progressbar.ProgressBar(widgets=['Running V2G-Sim: ', progressbar.Percentage(), progressbar.Bar()],
                                       maxval=(len(vehicleList))).start()

    for indexV, vehicle in enumerate(vehicleList):
        lastIteration = False

        for indexI in range(0, nbIteration):
            if indexI == nbIteration - 1:
                lastIteration = True

            for indexA, activity in enumerate(vehicle.itinerary[itineraryNb].activity):
                if isinstance(activity, model.Driving):
                    SOC, powerDemand, stranded = eval(vehicle.carModel.drivingFunc)
                    vehicle.SOC.extend(SOC)
                    if lastIteration:
                        activity.powerDemand.extend(powerDemand)
                        if stranded:
                            hourlySteps = 3600 / vehicle.outputInterval
                            vehicle.strandingLog.append(activity.start + (activity.end - activity.start)/2)

                elif isinstance(activity, model.Parked):
                    SOC, powerDemand = eval(activity.chargingInfra.chargingFunc)
                    vehicle.SOC.extend(SOC)
                    if lastIteration:
                        activity.powerDemand.extend(powerDemand)

                else:
                    print('Oops one of the activities was neither driving or parked')
                    return

            # Initiate Vehicle SOC last value to be the inital SOC next iteration
            if not lastIteration:
                vehicle.SOC = [vehicle.SOC[-1]]
            if lastIteration:
                del SOC[0]  # removed initial value added line 58

        progress.update(indexV + 1)
    progress.finish()
    print('')




[docs]def runV2GSim_reduce(vehicleList, nbIteration=1, option=None, itineraryNb=0):
    """ Launch driving functions and charging functions for each vehicle passed,
    results from function is then appended to the vehicle state. In order to overpass RAM limitation the results from
    individual activities are freed from the memory, instead a dictionary with the consumptions at each locations
    is returned.

    Args:
        vehicleList (list of vehicles): vehicles to simulate
        nbIteration (int): number of iterations
        option (anything): optional parameter for charging functions
        itineraryNb (int): optional itinerary number (default: 0)
    """
    print(colored.green('core.runV2GSim(vehicleList, nbIteration=' + str(nbIteration) + ')'))

    # Create the progress bar
    progress = progressbar.ProgressBar(widgets=['Running V2G-Sim: ', progressbar.Percentage(), progressbar.Bar()],
                                       maxval=(len(vehicleList))).start()
    windowTime = 24
    hourlySteps = 3600 / vehicleList[0].outputInterval
    powerDemandLoc = {'Total': [0] * (int(windowTime * hourlySteps))}
    for indexV, vehicle in enumerate(vehicleList):
        lastIteration = False

        for indexI in range(0, nbIteration):
            if indexI == nbIteration - 1:
                lastIteration = True

            for indexA, activity in enumerate(vehicle.itinerary[itineraryNb].activity):
                if isinstance(activity, model.Driving):
                    SOC, powerDemand, stranded = eval(vehicle.carModel.drivingFunc)
                    vehicle.SOC.extend(SOC)
                    if lastIteration:
                        if stranded:
                            hourlySteps = 3600 / vehicle.outputInterval
                            vehicle.strandingLog.append(activity.start + (activity.end - activity.start)/2)

                elif isinstance(activity, model.Parked):
                    SOC, powerDemand = eval(activity.location.chargingInfra.chargingFunc)
                    vehicle.SOC.extend(SOC)
                    if lastIteration:
                        if activity.pluggedIn:
                            # Set offset for the activity starting time
                            sublist = [0] * (int(hourlySteps * activity.start))
                            sublist.extend(powerDemand)
                            if activity.location.type not in powerDemandLoc:
                                powerDemandLoc[activity.location.type] = [0] * (int(windowTime * hourlySteps))

                            # Sum power demand at the corresponding location
                            powerDemandLoc[activity.location.type] = [sum(x) for x in itertools.izip_longest(
                                powerDemandLoc[activity.location.type], sublist, fillvalue=0)]
                            powerDemandLoc['Total'] = [sum(x) for x in
                                                       itertools.izip_longest(powerDemandLoc['Total'], sublist,
                                                                              fillvalue=0)]
            # Initiate Vehicle SOC last value to be the inital SOC next iteration
            if not lastIteration:
                vehicle.SOC = [vehicle.SOC[-1]]
            if lastIteration:
                del SOC[0]  # removed initial value added line 121

        progress.update(indexV + 1)
    progress.finish()
    print('')
    return powerDemandLoc
